PHP Composition, Dependency Injection and Polymorphism - Quick Table

Design / Pattern Code
1. Single Responsibility Principle broken 
<?php

class Song
{
    public function __construct(
        public string $title,
        public string $artist,
        public int $duration,
    ) {
    }

    public function writeSongData(): string
    {
        // Problem:
        // Song is responsible for song data AND text formatting.
        return "Title is " . $this->title .
            " Artist is " . $this->artist .
            " Duration is " . $this->duration;
    }

    public function writeJsonData(): string
    {
        // Problem:
        // Song is also responsible for JSON formatting.
        // This breaks Single Responsibility Principle.
        return json_encode($this);
    }
}

// Memory:
// One class should have one main reason to change.
// Here Song changes if song data changes OR output format changes.
2. Composition without dependency injection 
<?php

class Song
{
    private SongDataWriter $textWriter;
    private JsonDataWriter $jsonWriter;

    public function __construct(
        public string $title,
        public string $artist,
        public int $duration,
    ) {
        // No dependency injection:
        // Song creates its own dependency inside the constructor.
        $this->textWriter = new SongDataWriter();
        $this->jsonWriter = new JsonDataWriter();
    }

    public function writeText(): string
    {
        return $this->textWriter->write($this);
    }

    public function writeJson(): string
    {
        return $this->jsonWriter->write($this);
    }
}

// Memory:
// Composition = Song HAS writer objects. that is class made of other classes.
// No DI = Song creates those writer objects itself.
3. Composition with dependency injection 
<?php

class Song
{
    public function __construct(
        public string $title,
        public string $artist,
        public int $duration,
        private SongDataWriter $writer,
    ) {
        // Dependency injection:
        // The writer comes from outside the Song class.
    }

    public function write(): string
    {
        return $this->writer->write($this);
    }
}

$song = new Song(
    "I Disappear",
    "Metallica",
    300,
    new SongDataWriter()
);

// Memory:
// DI = do not create dependency inside the class.
// Pass the dependency into the class.
4. Polymorphism with interface 
<?php

interface DataWriterInterface
{
    public function write(Song $song): string;
}

class Song
{
    public function __construct(
        public string $title,
        public string $artist,
        public int $duration,
        private DataWriterInterface $writer,
    ) {
        // Song does not care which writer class it receives.
        // It only cares that the class follows the interface.
    }

    public function write(): string
    {
        return $this->writer->write($this);
    }
}

class SongDataWriter implements DataWriterInterface
{
    public function write(Song $song): string
    {
        return "Title is " . $song->title;
    }
}

class JsonDataWriter implements DataWriterInterface
{
    public function write(Song $song): string
    {
        return json_encode($song);
    }
}

$textSong = new Song("I Disappear", "Metallica", 300, new SongDataWriter());
$jsonSong = new Song("I Disappear", "Metallica", 300, new JsonDataWriter());

// Memory:
// Polymorphism = same method call, different behavior.
// Interface = classes promise to have the same method.
5. Polymorphism with inheritance 
<?php

abstract class BaseDataWriter
{
    abstract public function write(Song $song): string;
}

class Song
{
    public function __construct(
        public string $title,
        public string $artist,
        public int $duration,
        private BaseDataWriter $writer,
    ) {
        // Song accepts the parent type.
        // Any child of BaseDataWriter can be used here.
    }

    public function write(): string
    {
        return $this->writer->write($this);
    }
}

class SongDataWriter extends BaseDataWriter
{
    public function write(Song $song): string
    {
        return "Title is " . $song->title;
    }
}

class JsonDataWriter extends BaseDataWriter
{
    public function write(Song $song): string
    {
        return json_encode($song);
    }
}

$textSong = new Song("I Disappear", "Metallica", 300, new SongDataWriter());
$jsonSong = new Song("I Disappear", "Metallica", 300, new JsonDataWriter());

// Memory:
// Inheritance polymorphism = parent type can hold child objects.
// BaseDataWriter variable can receive SongDataWriter or JsonDataWriter.
